(5).add(3).reduce(2)

2019-07-10 #js基础 
Number.prototype.add = function(num) {
    if (typeof num === 'number') {
        return this + num;
    }
    return this;
}
Number.prototype.reduce = function(num) {
    if (typeof num === 'number') {
        return this - num;
    }
    return this;
}

console.log((5).add(3).reduce(2)); // 6

注意

这里使用 Number 和 Object 均可。
