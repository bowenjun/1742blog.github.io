【leetcode】反转链表

· 2019-07-29 · # 算法

反转一个单链表。

示例:

输入: 1->2->3->4->5->NULL
输出: 5->4->3->2->1->NULL

/**
 * Definition for singly-linked list.
 * function ListNode(val) {
 *     this.val = val;
 *     this.next = null;
 * }
 */
/**
 * @param {ListNode} head
 * @return {ListNode}
 */
// 常规
function reverseList1(head) {
	if (head === null || head.next === null) return head;
	let p = head.next;
	head.next = null;
	let temp = null;
	while (p !== null) {
		temp = p.next;
		p.next = head;
		head = p;
		p = temp;
	}
	return head;
}
// 递归
function reverseList2(head) {
	if (head === null || head.next === null) return head;
	let new_head = reverseList2(head.next);
	head.next.next = head;
	head.next = null;
	return new_head;
}
